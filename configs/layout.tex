% 页眉页脚相关
% 无页眉页脚（适用于封面页、独创性声明页）
\fancypagestyle{cover}
{
    \fancyhead{}
    \fancyfoot{}
    \renewcommand{\headrulewidth}{0pt}
}
% 适用于正文、参考文献的样式，使用阿拉伯数字表示页码
\fancypagestyle{body}
{
    \fancyhead{}
    \fancyhead[L]{\zihao{-5} 中国传媒大学硕士学位论文}
    \fancyhead[R]{\zihao{-5} \leftmark}
    \renewcommand{\headrulewidth}{0.6pt}

    % 注意切换下面两句以区分单双面打印
    % C表示居中
    % \fancyfoot[C]{\zihao{-5} \thepage}
    % OR表示奇数页居右，EL表示偶数页居左（下同）
    \fancyfoot[OR,EL]{\zihao{-5} \thepage}
}
% 适用于致谢、摘要、目录部分，使用罗马数字表示页码
\fancypagestyle{previous}
{
    \fancyhead{}
    \fancyhead[L]{\zihao{-5} 中国传媒大学硕士学位论文}
    \fancyhead[R]{\zihao{-5} \leftmark}
    \renewcommand{\headrulewidth}{0.6pt}

    % 注意切换下面两句以区分单双面打印
    % \fancyfoot[C]{\zihao{-5}\Roman{page}}
    \fancyfoot[OR,EL]{\zihao{-5}\Roman{page}}
}
