% 该文件设置代码块的样式

% 等宽字体样式可选
% \setmonofont{FiraCode}
\setmonofont{Consolas}

% 设置代码块样式
\lstset{
    columns=fixed,
    basicstyle=\footnotesize\ttfamily,                   % 设置字体      
    numbers=left,                                        % 在左侧显示行号
    keywordstyle=\color[RGB]{40,40,255},                 % 关键字样式
    numberstyle=\footnotesize\color{black},              % 行号样式
    commentstyle=\it\color[RGB]{0,96,96},                % 设置代码注释的格式
    stringstyle=\rmfamily\slshape\color[RGB]{128,0,0},   % 设置字符串格式
    showstringspaces=false,                              % 不显示字符串中的空格
    frame=shadowbox,                                     % 边框样式
    breaklines=true,                                     % 换行相关
    breakatwhitespace=true,                              % 换行相关
    language=c++,                                        % 默认语言（可高亮显示关键字）
}
