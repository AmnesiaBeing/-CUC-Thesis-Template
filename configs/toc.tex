\renewcommand{\cftloftitlefont}{\hfill\zihao{3}\bfseries}\renewcommand{\cftafterloftitle}{\hfill}

% 目录页各级字体
\renewcommand{\cftchapfont}         {\zihao{4}\heiti}
\renewcommand{\cftchapleader}       {\cftdotfill{\cftchapdotsep}}
\renewcommand{\cftchappagefont}     {}

\renewcommand{\cftsecnumwidth}        {3em}
\renewcommand{\cftsubsecnumwidth}     {3em}
\renewcommand{\cftsubsubsecnumwidth}  {3em}

\cftsetindents{figure}{0em}{2em}
\cftsetindents{table}{0em}{2em}

\setlength{\cftsubsubsecindent} {0em}
